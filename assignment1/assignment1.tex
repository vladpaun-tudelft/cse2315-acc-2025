	\subsection{Definitions}
	\begin{definition}
		A function $f : \N \to \N$ is \emph{monotone} if $f(n+1) \ge f(n)$.
	\end{definition}

	\subsection{Lemma Example}
	\begin{lemma}
		For all $n \in \N$, $n^2 \ge n$.
	\end{lemma}

	\begin{proof}
		For $n \ge 1$, we have $n^2 - n = n(n-1) \ge 0$.
	\end{proof}
